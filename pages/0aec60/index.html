#include <stdio.h>
#include <stdlib.h>
#include <string.h>
char buf[32];
// argc 是命令行参数的数量。
// argv 是一个字符串数组，包含命令行参数。
// envp 是一个环境变量的字符串数组
int main(int argc, char* argv[], char* envp[]){
	if(argc<2){
		printf("pass argv[1] a number\n");
		return 0;
	}
	int fd = atoi( argv[1] ) - 0x1234;
	int len = 0;
    // fd=0 表示标准输入
    // 1 表示标准输出
    // 2 表示错误输出
	len = read(fd, buf, 32);
	if(!strcmp("LETMEWIN\n", buf)){
		printf("good job :)\n");
		system("/bin/cat flag");
		exit(0);
	}
	printf("learn about Linux file IO\n");
	return 0;
}
